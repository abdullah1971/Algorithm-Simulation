 void mergesort(int *arr, int low, int high)
 {
     int mid;
     if (low < high)
     {
         mid=(low+high)/2;
         mergesort(arr,low,mid);
         mergesort(arr,mid+1,high);
         merge(arr,low,high,mid);
     }
     return;
 }
 
 
 void merge(int *arr, int low, int high, int mid)
 {
     int i, j, k, temp[50];
     i = low;
     k = low;
     j = mid + 1;
     while (i <= mid && j <= high)
     {
         if (arr[i] < arr[j])
        {
             temp[k] = arr[i];
             k++;
             i++;
         }
         else
         {
             temp[k] = arr[j];
             k++;
             j++;
         }
     }
     while (i <= mid)
     {
         temp[k] = arr[i];
         k++;
         i++;
     }
     while (j <= high)
     {
         temp[k] = arr[j];
         k++;
         j++;
     }
     for (i = low; i < k; i++)
     {
         arr[i] = temp[i];
     }
 }
